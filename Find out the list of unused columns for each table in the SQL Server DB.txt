--- Find out the list of unused columns for each table in the SQL Server DB
-- Short query to find unused columns in SQL Server tables
WITH UsedColumns AS (
    SELECT DISTINCT
        OBJECT_SCHEMA_NAME(c.object_id) AS SchemaName,
        OBJECT_NAME(c.object_id) AS TableName,
        c.name AS ColumnName
    FROM sys.columns c
    WHERE c.object_id IN (SELECT object_id FROM sys.tables)
    AND (
        -- Column is indexed
        EXISTS (SELECT 1 FROM sys.index_columns ic WHERE ic.object_id = c.object_id AND ic.column_id = c.column_id)
        OR
        -- Column is in foreign key
        EXISTS (SELECT 1 FROM sys.foreign_key_columns fkc WHERE 
                (fkc.parent_object_id = c.object_id AND fkc.parent_column_id = c.column_id) OR
                (fkc.referenced_object_id = c.object_id AND fkc.referenced_column_id = c.column_id))
        OR
        -- Column is identity or computed
        c.is_identity = 1 OR c.is_computed = 1
    )
)
SELECT 
    s.name AS SchemaName,
    t.name AS TableName,
    c.name AS ColumnName,
    ty.name AS DataType,
    c.max_length,
    CASE WHEN c.is_nullable = 1 THEN 'YES' ELSE 'NO' END AS IsNullable,
    'ALTER TABLE [' + s.name + '].[' + t.name + '] DROP COLUMN [' + c.name + '];' AS DropStatement
FROM sys.tables t
INNER JOIN sys.schemas s ON s.schema_id = t.schema_id
INNER JOIN sys.columns c ON c.object_id = t.object_id
INNER JOIN sys.types ty ON ty.system_type_id = c.system_type_id AND ty.user_type_id = c.user_type_id
LEFT JOIN UsedColumns uc ON s.name = uc.SchemaName AND t.name = uc.TableName AND c.name = uc.ColumnName
WHERE uc.ColumnName IS NULL  -- Column not found in used columns
ORDER BY s.name, t.name, c.name;

--- Even shorter version (basic check)

SELECT 
    OBJECT_SCHEMA_NAME(c.object_id) AS SchemaName,
    OBJECT_NAME(c.object_id) AS TableName,
    c.name AS ColumnName,
    'ALTER TABLE [' + OBJECT_SCHEMA_NAME(c.object_id) + '].[' + OBJECT_NAME(c.object_id) + '] DROP COLUMN [' + c.name + '];' AS DropStatement
FROM sys.columns c
INNER JOIN sys.tables t ON t.object_id = c.object_id
WHERE c.is_identity = 0 
  AND c.is_computed = 0
  AND NOT EXISTS (SELECT 1 FROM sys.index_columns ic WHERE ic.object_id = c.object_id AND ic.column_id = c.column_id)
  AND NOT EXISTS (SELECT 1 FROM sys.foreign_key_columns fkc WHERE 
                  (fkc.parent_object_id = c.object_id AND fkc.parent_column_id = c.column_id) OR
                  (fkc.referenced_object_id = c.object_id AND fkc.referenced_column_id = c.column_id))
ORDER BY OBJECT_SCHEMA_NAME(c.object_id), OBJECT_NAME(c.object_id), c.name;
